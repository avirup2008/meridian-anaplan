      const panel = document.getElementById('evidence-limit-warning');
      const list = document.getElementById('blocked-conclusions-list');
      if (!panel || !list) return;
      const blocked = (evidencePack && Array.isArray(evidencePack.blockedConclusions)) ? evidencePack.blockedConclusions : [];
      if (!blocked.length) {
        panel.hidden = true;
        list.innerHTML = '';
        return;

| T-06-18 | Information Disclosure | Credentials in DevTools network tab | accept | Same exposure as v2.0; password is only in memory during the active session; HTTPS in transit; sessionStorage cleared on browser close |
