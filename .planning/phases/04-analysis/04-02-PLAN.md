    // SECTION: ANALYSIS
    // Phase 4 — consumes /api/analyze SSE stream; renders score card, suggestions, narrative + drill-in.

    let _anlState = null;   // accumulated analysis state across SSE events
    let _anlRunning = false;

    function _anl$(id) { return document.getElementById(id); }

    function _anlReset() {
      _anlState = {
        suggestionsByDomain: { 'Structural': [], 'Formula': [], 'Best Practice': [], 'Naming': [] },
        suggestionCount: 0,
        modules: [],     // narrative modules
        score: null,
        partial: null,
      };
      _anl$('anl-error').hidden = true;
      _anl$('anl-warning').hidden = true;
      _anl$('anl-score-card').hidden = true;
      _anl$('anl-suggestions').hidden = true;
      _anl$('anl-narrative').hidden = true;
      _anl$('anl-progress-wrap').style.display = '';
      _anl$('anl-progress-bar').style.width = '0%';
      _anl$('anl-progress-pct').textContent = '0%';
      _anl$('anl-stage-label').textContent = 'Starting…';
      _anl$('anl-progress-detail').textContent = ' ';
      _anl$('anl-suggestion-domains').innerHTML = '';
      _anl$('anl-story-text').textContent = '';
      _anl$('anl-module-nodes').innerHTML = '';
      _anl$('anl-narrative-story-view').hidden = false;
      _anl$('anl-narrative-drill-view').hidden = true;
    }

    function _anlShowError(message) {
      _anl$('anl-error').hidden = false;
      _anl$('anl-error-msg').textContent = message || 'Unknown error';
      _anl$('anl-progress-wrap').style.display = 'none';
      _anl$('anl-hdr-status').innerHTML = 'Analysis failed';
    }

    function _anlOnProgress(stage, pct) {
      const labels = {
        fetching: 'Fetching blueprint',
        extracting: 'Extracting modules',
        suggestions: 'Analysing modules (Haiku)',
        scoring: 'Computing health score (Sonnet)',
        narrative: 'Generating data-flow narrative',
      };
      _anl$('anl-stage-label').textContent = labels[stage] || stage;
      _anl$('anl-progress-bar').style.width = pct + '%';
      _anl$('anl-progress-pct').textContent = pct + '%';
    }

    function _anlOnExtractionDone(moduleCount, totalLineItems) {
      _anl$('anl-progress-detail').textContent = moduleCount + ' modules, ' + totalLineItems + ' line items extracted';
    }

    function _anlOnHaikuProgress(done, total, name, skipped) {
      _anl$('anl-progress-detail').textContent = 'Analysed ' + done + ' / ' + total + (skipped ? ' (skipped: ' + name + ')' : ' — ' + name);
      // Visual sub-progress within the "suggestions" stage (25-65%)
      const pct = 25 + Math.round((done / Math.max(total, 1)) * 40);
      _anl$('anl-progress-bar').style.width = pct + '%';
      _anl$('anl-progress-pct').textContent = pct + '%';
    }

    function _anlOnSuggestions(evt) {
      if (!Array.isArray(evt.items)) return;
      for (const it of evt.items) {
        if (!_anlState.suggestionsByDomain[it.domain]) continue;
        _anlState.suggestionsByDomain[it.domain].push({ ...it, moduleId: evt.moduleId, moduleName: evt.moduleName });
        _anlState.suggestionCount++;

    async function runAnalysis() {
      if (_anlRunning) return;
      _anlRunning = true;
      _anlReset();

      const blobUrl = sessionStorage.getItem('meridian.blueprintBlobUrl');
      if (!blobUrl) {
        _anlShowError('No blueprint URL found. Please fetch a blueprint first.');
        _anlRunning = false;
        return;
      }

      try {
        const resp = await fetch('/api/analyze', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ blobUrl }),
        });
        if (!resp.ok || !resp.body) {
          const errText = await resp.text().catch(() => '');
          throw new Error('HTTP ' + resp.status + (errText ? ': ' + errText.slice(0, 200) : ''));
        }
        const reader = resp.body.getReader();
        const decoder = new TextDecoder();
        let buf = '';
        while (true) {
          const { done, value } = await reader.read();
          if (done) break;
          buf += decoder.decode(value, { stream: true });
          const lines = buf.split('\n');
          buf = lines.pop() || '';
          for (const line of lines) {
            if (!line.startsWith('data:')) continue;
            const payload = line.slice(5).trim();
            if (!payload) continue;
            let evt;
            try { evt = JSON.parse(payload); } catch { continue; }
            switch (evt.type) {
              case 'progress': _anlOnProgress(evt.stage, evt.pct); break;
              case 'extraction-done': _anlOnExtractionDone(evt.moduleCount, evt.totalLineItems); break;
              case 'haiku-progress': _anlOnHaikuProgress(evt.modulesDone, evt.modulesTotal, evt.moduleName, evt.skipped); break;
              case 'suggestions': _anlOnSuggestions(evt); break;
              case 'partial-analysis': _anlOnPartial(evt.reason, evt.modulesAnalysed, evt.modulesSkipped); break;
              case 'score': _anlOnScore(evt); break;
              case 'narrative': _anlOnNarrative(evt); break;
              case 'complete': _anlOnComplete(evt); break;
              case 'error': _anlShowError(evt.message); break;
            }
          }
        }
      } catch (err) {
        _anlShowError(err && err.message ? err.message : String(err));
      } finally {
        _anlRunning = false;
      }
    }

        const r = _origGo.apply(this, arguments);
