// Deterministic health score base тАФ finding severity weighted by host module blast radius (D-06).
function computeDeterministicHealthScore(findings, blastRadiusById, moduleCount) {
  const SEVERITY_WEIGHT = { critical: 4, warning: 2, info: 1 };
  let penalty = 0;
  for (const f of findings) {
    const w = SEVERITY_WEIGHT[f.severity] || 1;
    const blast = blastRadiusById.get(f.moduleId) || 0;
    penalty += w * (1 + blast * 0.25);
  }
  // Normalize: at penalty ~ moduleCount * 4, score тЖТ 0. Scale roughly so a clean model = ~95.
  const normalized = Math.max(0, 100 - (penalty / Math.max(moduleCount, 1)) * 8);
  return Math.round(Math.min(95, normalized));
}
