```javascript
function _anlOnHealthWorkstreams(evt) {
  // Always render honest limits (D-04)
  const limitsBox = _anl$('health-honest-limits');
  const limitsList = _anl$('health-honest-limits-list');
  if (limitsBox && limitsList && Array.isArray(evt.honestLimits) && evt.honestLimits.length) {
    limitsList.innerHTML = evt.honestLimits.map(h => '<li>' + escapeHtml(h) + '</li>').join('');
    limitsBox.hidden = false;
  }
  // Architecture verdict + score card (D-06, D-07)
  const verdictCard = _anl$('health-verdict-card');
  if (verdictCard) {
    if (evt.architectureVerdict) _anl$('health-verdict-text').textContent = evt.architectureVerdict;
    if (Number.isFinite(evt.healthScore)) {
      _anl$('health-score-num').textContent = String(evt.healthScore);
    }
    const split = _anl$('health-score-split');
    if (split && Number.isFinite(evt.healthScoreDeterministic) && Number.isFinite(evt.healthScoreSonnet)) {
      split.textContent = 'rules: ' + evt.healthScoreDeterministic + ' · review: ' + evt.healthScoreSonnet;
    }
    const reasoning = _anl$('health-score-reasoning');
    if (reasoning && evt.healthScoreReasoning) reasoning.textContent = evt.healthScoreReasoning;
    verdictCard.hidden = false;
  }
  // Format-specific render
  const slot = _anl$('health-format-slot');
  if (slot) {
    slot.innerHTML = _renderHealthFormat(evt);
    slot.hidden = false;
  }
  // Evidence limits (existing canSay/cannotSay) — KEEP existing render block at the bottom
  // (existing code from current implementation preserved below this point)
  const evLimits = _anl$('health-evidence-limits');
  if (evLimits && evt.evidenceLimits) {
    const canSayEl = evLimits.querySelector('[data-key="canSay"]');
    const cannotSayEl = evLimits.querySelector('[data-key="cannotSay"]');
    if (canSayEl) canSayEl.innerHTML = (evt.evidenceLimits.canSay || []).map(s => '<li style="margin-bottom:4px;">' + escapeHtml(s) + '</li>').join('');
    if (cannotSayEl) cannotSayEl.innerHTML = (evt.evidenceLimits.cannotSay || []).map(s => '<li style="margin-bottom:4px;">' + escapeHtml(s) + '</li>').join('');
    evLimits.hidden = false;
  }
}

function _renderHealthFormat(evt) {
  const fmt = evt.format;
  if (fmt === 'surgical' && evt.surgical?.findings) {
    return '<h3 style="margin:18px 0 8px;">Findings (' + evt.surgical.findings.length + ')</h3>'
      + evt.surgical.findings.map(f => {
          const blast = Number.isFinite(f.downstream) ? f.downstream : 0;
          const blastStyle = blast > 10 ? 'color:#c41a1a;font-weight:700;' : 'color:#666;';
          return '<div style="padding:10px 12px;border-bottom:1px solid #eee;font-size:13px;">'
            + '<strong>' + escapeHtml(f.module || '?') + '</strong>'
            + ' · ' + escapeHtml(f.lineItem || '?')
            + ' · ' + escapeHtml(f.issue || '?')
            + ' · <span style="' + blastStyle + '">' + blast + ' downstream</span>'
            + (f.fix ? '<div style="margin-top:4px;font-size:12px;color:#555;">Fix: ' + escapeHtml(f.fix) + '</div>' : '')
            + '</div>';
        }).join('');
  }
  if (fmt === 'brief' && evt.brief) {
    const b = evt.brief;
    let html = '';
    if (b.domainCoverage) html += '<h3>Domain Coverage</h3><p>' + escapeHtml(b.domainCoverage) + '</p>';
    if (b.architectureShape) html += '<h3>Architecture Shape</h3><p>' + escapeHtml(b.architectureShape) + '</p>';
    if (Array.isArray(b.top3Risks) && b.top3Risks.length) {
      html += '<h3>Top 3 Named Risks</h3><ol>';
      for (const r of b.top3Risks) {
        html += '<li><strong>' + escapeHtml(r.module || '?') + '</strong> — '
          + escapeHtml(r.lineItem || '?') + ' · ' + escapeHtml(r.issue || '?')
          + ' (' + (Number(r.downstream) || 0) + ' downstream)</li>';
      }
      html += '</ol>';
    }
    return html;
  }
  if (fmt === 'domain' && evt.domain) {
    const d = evt.domain;
    let html = '';
    for (const dom of (d.domains || [])) {
      html += '<h3>' + escapeHtml(dom.name || 'Domain') + '</h3>';
      html += '<p style="font-size:13px;color:#555;">' + (dom.moduleCount || 0) + ' modules. ' + escapeHtml(dom.description || '') + '</p>';
      if (Array.isArray(dom.findings)) {
        html += '<ul style="font-size:13px;">' + dom.findings.map(f =>
          '<li>' + escapeHtml(f.module || '?') + ' · ' + escapeHtml(f.lineItem || '?') + ' · ' + escapeHtml(f.issue || '?') + ' · ' + (Number(f.downstream) || 0) + ' downstream</li>'
        ).join('') + '</ul>';
      }
    }
    if (Array.isArray(d.integrationSeams) && d.integrationSeams.length) {
      html += '<h3>Integration Seams</h3><ul style="font-size:13px;">'
        + d.integrationSeams.map(s => '<li>' + escapeHtml(s.module || '?') + ' · ' + (Number(s.downstream) || 0) + ' downstream</li>').join('')
        + '</ul>';
    }
    return html;
  }
  // 'workstreams' format or fallback — render the existing card style
  const ws = (evt.workstreams || (evt.workstreams && evt.workstreams.workstreams) || []);
  if (!Array.isArray(ws) || !ws.length) return '<p style="color:#888;font-size:13px;">No workstreams available.</p>';
  return ws.map(w => {
    return '<div style="border:1px solid #e0e0e0;border-radius:6px;padding:14px 18px;margin:10px 0;">'
      + '<strong>' + escapeHtml(w.title || '?') + '</strong>'
      + ' <span style="font-size:11px;color:#666;">[' + escapeHtml(w.priority || '?') + ' · ' + escapeHtml(w.confidence || '?') + ']</span>'
      + '<p style="margin:6px 0 0;font-size:13px;color:#555;">' + escapeHtml(w.whyItMatters || '') + '</p>'
      + (w.examples?.length ? '<p style="margin:6px 0 0;font-size:11px;color:#aaa;">Examples: ' + w.examples.map(escapeHtml).join('; ') + '</p>' : '')
      + '</div>';
  }).join('');
}
```
