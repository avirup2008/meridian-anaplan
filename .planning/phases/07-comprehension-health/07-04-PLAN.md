function _anlOnModelComprehension(evt) {
  // Hide loading placeholder
  const loading = _anl$('model-loading');
  if (loading) loading.hidden = true;

  // ── Module classification table ──────────────────────────────────────────
  const classSection = _anl$('model-classification-section');
  const classBody = _anl$('model-classification-body');
  const classSubtitle = _anl$('model-classification-subtitle');
  if (classSection && classBody && evt.modules && evt.modules.length) {
    classSubtitle.textContent = `${evt.modules.length} functional modules classified`;
    classBody.innerHTML = '';
    for (const m of evt.modules) {
      const tr = document.createElement('tr');
      const confColor = m.confidenceLabel === 'High' ? '#2ecc71' : m.confidenceLabel === 'Medium' ? '#f39c12' : '#e74c3c';
      tr.innerHTML = `
        <td style="padding:6px 10px;border-bottom:1px solid #eee;">${escapeHtml(m.moduleName)}</td>
        <td style="padding:6px 10px;border-bottom:1px solid #eee;font-family:monospace;">${escapeHtml(m.discoLabel)}</td>
        <td style="padding:6px 10px;border-bottom:1px solid #eee;">
          <span style="display:inline-block;padding:2px 8px;border-radius:12px;background:${confColor}22;color:${confColor};font-size:11px;font-weight:600;" title="${escapeHtml(m.confidenceReason)}">${escapeHtml(m.confidenceLabel)}</span>
        </td>
        <td style="padding:6px 10px;border-bottom:1px solid #eee;text-align:right;">${m.formulaCount}</td>
        <td style="padding:6px 10px;border-bottom:1px solid #eee;text-align:right;">${m.inputCount}</td>
      `;
      classBody.appendChild(tr);
    }
    classSection.hidden = false;
  }

  // ── DISCO architecture map ────────────────────────────────────────────────
  const discoSection = _anl$('model-disco-section');
  const discoMap = _anl$('model-disco-map');
  if (discoSection && discoMap && evt.discoMap) {
    discoMap.innerHTML = '';
    const labels = { SYS: 'System', DAT: 'Data', CAL: 'Calculation', INP: 'Planning', REP: 'Output', UNKNOWN: 'Unknown' };
    for (const [key, count] of Object.entries(evt.discoMap)) {
      if (!count && key !== 'UNKNOWN') continue;
      const tile = document.createElement('div');
      tile.className = 'intel-card';
      tile.style.cssText = 'min-width:100px;text-align:center;padding:14px 18px;';
      if (key === 'UNKNOWN' && count > 0) tile.style.borderColor = '#e74c3c';
      tile.innerHTML = `<div style="font-size:22px;font-weight:700;">${count}</div><div style="font-size:11px;margin-top:4px;opacity:.7;">${key} — ${labels[key] || key}</div>`;
      discoMap.appendChild(tile);
    }
    discoSection.hidden = false;
  }

  // ── Limitation cards (when gates fail) ───────────────────────────────────
  const limSection = _anl$('model-limitations-section');
  const limCards = _anl$('model-limitation-cards');
  if (limSection && limCards && evt.limitationCards && evt.limitationCards.length) {
    limCards.innerHTML = '';
    for (const text of evt.limitationCards) {
      const card = document.createElement('div');
      card.className = 'limitation-card';
      card.style.cssText = 'background:#fff8f0;border:1px solid #f39c12;border-radius:6px;padding:12px 16px;margin-bottom:8px;';
      card.innerHTML = `<strong style="color:#b7770d;">Limitation</strong><p style="margin:6px 0 0;font-size:13px;">${escapeHtml(text)}</p>`;
      limCards.appendChild(card);
    }
    limSection.hidden = false;
  }

  // ── Dependency graph table ────────────────────────────────────────────────
  const graphSection = _anl$('model-graph-section');
  const graphBody = _anl$('model-graph-body');
  const graphSubtitle = _anl$('model-graph-subtitle');
  if (graphSection && graphBody && evt.graph && evt.graph.edges && evt.graph.edges.length) {
    const edges = evt.graph.edges.slice().sort((a, b) => b.lineItems.length - a.lineItems.length);
    graphSubtitle.textContent = `${edges.length} cross-module reference edge${edges.length === 1 ? '' : 's'} detected`;
    graphBody.innerHTML = '';
    for (const edge of edges.slice(0, 100)) { // cap at 100 rows for rendering performance
      const tr = document.createElement('tr');
      tr.innerHTML = `
        <td style="padding:6px 10px;border-bottom:1px solid #eee;font-size:12px;">${escapeHtml(edge.fromModuleName)}</td>
        <td style="padding:6px 10px;border-bottom:1px solid #eee;font-size:12px;">${escapeHtml(edge.toModuleName)}</td>
        <td style="padding:6px 10px;border-bottom:1px solid #eee;text-align:right;font-size:12px;">${edge.lineItems.length}</td>
      `;
      graphBody.appendChild(tr);
    }
    graphSection.hidden = false;
  } else if (graphSection && evt.limitationCards && evt.limitationCards.length) {
    // Gates failed — graph already suppressed via limitationCards, don't show empty table
    graphSection.hidden = true;
  }

  // ── Dead logic, cycles, daisy chains ─────────────────────────────────────
  const patternsSection = _anl$('model-patterns-section');
  const deadLogicWrap = _anl$('model-dead-logic-wrap');
  const cyclesWrap = _anl$('model-cycles-wrap');
  const daisyWrap = _anl$('model-daisy-chains-wrap');
  let hasPatterns = false;

  if (deadLogicWrap && evt.deadLogic && evt.deadLogic.length) {
    hasPatterns = true;
    deadLogicWrap.innerHTML = `
      <details>
        <summary style="cursor:pointer;font-weight:600;font-size:13px;">
          Dead Logic — ${evt.deadLogic.length} unreferenced CALC item${evt.deadLogic.length === 1 ? '' : 's'}
          <span style="font-size:11px;font-weight:400;color:#888;margin-left:8px;">(Medium confidence — truncated formulas excluded)</span>
        </summary>
        <ul style="margin:8px 0 0 18px;font-size:12px;">
          ${evt.deadLogic.map(d => `<li><strong>${escapeHtml(d.moduleName)}</strong> / ${escapeHtml(d.lineItemName)}</li>`).join('')}
        </ul>
      </details>
    `;
  }

  if (cyclesWrap && evt.cycles && evt.cycles.length) {
    hasPatterns = true;
    const nodeById = new Map((evt.graph.nodes || []).map(n => [n.moduleId, n.moduleName]));
    cyclesWrap.innerHTML = `
      <details>
        <summary style="cursor:pointer;font-weight:600;font-size:13px;color:#c0392b;">
          Circular Dependencies — ${evt.cycles.length} cycle${evt.cycles.length === 1 ? '' : 's'} detected
        </summary>
        <ul style="margin:8px 0 0 18px;font-size:12px;">
          ${evt.cycles.map(c => `<li>${c.map(id => escapeHtml(nodeById.get(id) || id)).join(' → ')}</li>`).join('')}
        </ul>
      </details>
    `;
  }

  if (daisyWrap && evt.daisyChains && evt.daisyChains.length) {
    hasPatterns = true;
    const nodeById = new Map((evt.graph.nodes || []).map(n => [n.moduleId, n.moduleName]));
    daisyWrap.innerHTML = `
      <details>
        <summary style="cursor:pointer;font-weight:600;font-size:13px;">
          Single-Path Passthrough Modules — ${evt.daisyChains.length} candidate${evt.daisyChains.length === 1 ? '' : 's'}
          <span style="font-size:11px;font-weight:400;color:#888;margin-left:8px;">(review intent with model owner)</span>
        </summary>
        <ul style="margin:8px 0 0 18px;font-size:12px;">
          ${evt.daisyChains.map(id => `<li>${escapeHtml(nodeById.get(id) || id)}</li>`).join('')}
        </ul>
      </details>
    `;
  }

  if (patternsSection) patternsSection.hidden = !hasPatterns;
}

// XSS helper — used by model comprehension renderer and health workstream renderer
function escapeHtml(str) {
  return String(str || '').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

function _anlOnHealthWorkstreams(evt) {
  // ── Workstream cards ──────────────────────────────────────────────────────
  const wsSection = _anl$('health-workstreams-section');
  const wsList = _anl$('health-workstreams-list');
  const execBrief = _anl$('health-executive-brief');

  if (wsSection && wsList && evt.workstreams && evt.workstreams.length) {
    // Executive brief
    if (execBrief && evt.executiveBrief) {
      execBrief.textContent = evt.executiveBrief.split('\n\n')[0]; // first paragraph only for summary
    }

    wsList.innerHTML = '';
    const priorityColor = { Critical: '#c0392b', High: '#e67e22', Medium: '#2980b9', Watch: '#7f8c8d' };
    const kindBadge = { 'evidence-limit': '<span style="font-size:10px;background:#fff3cd;color:#856404;border-radius:10px;padding:2px 8px;margin-left:8px;">Evidence Limit</span>', 'remediation': '' };

    for (const ws of evt.workstreams) {
      const color = priorityColor[ws.priority] || '#7f8c8d';
      const confColor = ws.confidence === 'High' ? '#2ecc71' : ws.confidence === 'Medium' ? '#f39c12' : '#e74c3c';
      const card = document.createElement('div');
      card.className = 'intel-card wide';
      card.style.cssText = 'margin-bottom:12px;border-left:4px solid ' + color + ';';
      card.innerHTML = `
        <div style="display:flex;align-items:flex-start;justify-content:space-between;gap:12px;">
          <div>
            <strong style="font-size:14px;">${escapeHtml(ws.title)}${kindBadge[ws.kind] || ''}</strong>
            <p style="margin:6px 0 0;font-size:13px;color:#555;">${escapeHtml(ws.whyItMatters)}</p>
          </div>
          <div style="text-align:right;flex-shrink:0;">
            <div style="font-size:11px;font-weight:600;color:${color};white-space:nowrap;">${escapeHtml(ws.priority)}</div>
            <div style="font-size:11px;color:${confColor};white-space:nowrap;margin-top:2px;">${escapeHtml(ws.confidence)} confidence</div>
            <div style="font-size:11px;color:#888;white-space:nowrap;margin-top:2px;">${ws.evidenceCount} finding${ws.evidenceCount === 1 ? '' : 's'}</div>
          </div>
        </div>
        ${ws.reviewQuestion ? `<p style="margin:10px 0 0;font-size:12px;color:#888;font-style:italic;">Review: ${escapeHtml(ws.reviewQuestion)}</p>` : ''}
        ${ws.examples && ws.examples.length ? `<p style="margin:8px 0 0;font-size:11px;color:#aaa;">Examples: ${ws.examples.map(escapeHtml).join('; ')}</p>` : ''}
      `;
      wsList.appendChild(card);
    }
    wsSection.hidden = false;
  }

  // ── Evidence limits ───────────────────────────────────────────────────────
  const limSection = _anl$('health-evidence-limits');
  const canSayEl = _anl$('health-can-say');
  const cannotSayEl = _anl$('health-cannot-say');
  if (limSection && canSayEl && cannotSayEl && evt.evidenceLimits) {
    canSayEl.innerHTML = (evt.evidenceLimits.canSay || []).map(s => `<li style="margin-bottom:4px;">${escapeHtml(s)}</li>`).join('');
    cannotSayEl.innerHTML = (evt.evidenceLimits.cannotSay || []).map(s => `<li style="margin-bottom:4px;">${escapeHtml(s)}</li>`).join('');
    limSection.hidden = false;
  }
